//Author:PanDaoxi
#include <iostream>
using namespace std;
const int n=100;
int s[n+1], // 模拟栈 
top=0; // 指针 
void push(int x){ // 入栈 
	if(top<n){
		top++; // 指针自增 
		s[top]=x; // 元素入栈 
		return;
	}
}
void pop(){ // 出栈 
	if(top>0) top--; //指针自减 
	return; 
}
void clear(){ //清空栈 
	top=0; // 指针清空 
	return;
} 
void gtop(){ // 获取栈顶元素 
	cout<<"栈顶的元素："<<s[top]<<endl; 
	return; 
}
int main(){
	// 1.入栈：1 2 3
	push(1);
	push(2);
	push(3);
	cout<<"入栈后元素： ";
	for(int i=1;i<=top;i++){
		cout<<s[i]<<" ";
	} 
	cout<<endl;
	// 2.出栈：3
	pop(); 
	cout<<"出栈后元素： ";
	for(int i=1;i<=top;i++){
		cout<<s[i]<<" ";
	} 
	cout<<endl;
	// 3.获取栈顶元素：2 
	gtop();
	// 4.清空栈 
	clear();
	cout<<"清空栈： ";
	for(int i=1;i<=top;i++){
		cout<<s[i]<<" ";
	} 
	cout<<endl;
	return 0;
}
