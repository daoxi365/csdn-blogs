```python
#设置字典
school = {
		'五年级（1）班' : ('小赵','小钱','小孙','小李'),
		'五年级（2）班' : ('小周','小吴','小郑','小王'),
		'五年级（3）班' : ('小冯','小陈','小褚','小卫'),
		'五年级（4）班' : ('小蒋','小沈','小韩','小杨')
		}

#循环
for i in range(1,5):
	for student in school['五年级（' + str(i) + '）班']:
		print(student)
```
